## –°–æ–±—Å—Ç–≤–µ–Ω–Ω–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è –¥–µ–∫–∞ –Ω–∞ –¥–≤—É—Å–≤—è–∑–Ω–æ–º —Å–ø–∏—Å–∫–µ (Python)

```python
class Node:
    def __init__(self, data):
        self.data = data
        self.prev = None
        self.next = None

class Deque:
    def __init__(self):
        self.front = None
        self.rear = None
        self._size = 0

    def push_front(self, data):
        new_node = Node(data)
        if not self.front:
            self.front = self.rear = new_node
        else:
            new_node.next = self.front
            self.front.prev = new_node
            self.front = new_node
        self._size += 1

    def push_back(self, data):
        new_node = Node(data)
        if not self.rear:
            self.front = self.rear = new_node
        else:
            new_node.prev = self.rear
            self.rear.next = new_node
            self.rear = new_node
        self._size += 1

    def pop_front(self):
        if not self.front:
            print("Deque is empty")
            return None
        data = self.front.data
        self.front = self.front.next
        if self.front:
            self.front.prev = None
        else:
            self.rear = None
        self._size -= 1
        return data

    def pop_back(self):
        if not self.rear:
            print("Deque is empty")
            return None
        data = self.rear.data
        self.rear = self.rear.prev
        if self.rear:
            self.rear.next = None
        else:
            self.front = None
        self._size -= 1
        return data

    def peek_front(self):
        return None if not self.front else self.front.data

    def peek_back(self):
        return None if not self.rear else self.rear.data

    def is_empty(self):
        return self._size == 0

    def size(self):
        return self._size
```
