```python
def shell_sort(arr: list[int]) -> None:
    n = len(arr)
    gap = n // 2

    while gap > 0:
        for i in range(gap, n):
            temp = arr[i]
            j = i

            # —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –≤—Å—Ç–∞–≤–∫–∞–º–∏ —Å —à–∞–≥–æ–º gap

                arr[j] = arr[j - gap]
                j -= gap

            arr[j] = temp

        gap //= 2
```
