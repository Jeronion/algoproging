## –†–µ–∞–ª–∏–∑–∞—Ü–∏—è –∞–ª–≥–æ—Ä–∏—Ç–º–∞ (Python)

```python

def countingSort(arr, exp1):

    n = len(arr)

    # –í—ã—Ö–æ–¥–Ω–æ–π –º–∞—Å—Å–∏–≤
    output = [0] * n

    # –ú–∞—Å—Å–∏–≤ –¥–ª—è –ø–æ–¥—Å—á—ë—Ç–∞ —Ü–∏—Ñ—Ä (0‚Äì9)
    count = [0] * 10

    # –ü–æ–¥—Å—á—ë—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –≤—Ö–æ–∂–¥–µ–Ω–∏–π —Ü–∏—Ñ—Ä
    for i in range(n):
        index = arr[i] // exp1
        count[index % 10] += 1

    # –ò–∑–º–µ–Ω—è–µ–º count[i], —á—Ç–æ–±—ã –æ–Ω —Å–æ–¥–µ—Ä–∂–∞–ª
    # —Ñ–∞–∫—Ç–∏—á–µ—Å–∫—É—é –ø–æ–∑–∏—Ü–∏—é —Ü–∏—Ñ—Ä—ã –≤ output
    for i in range(1, 10):
        count[i] += count[i - 1]

    # –§–æ—Ä–º–∏—Ä—É–µ–º –≤—ã—Ö–æ–¥–Ω–æ–π –º–∞—Å—Å–∏–≤ (—Å–ø—Ä–∞–≤–∞ –Ω–∞–ª–µ–≤–æ ‚Äî –¥–ª—è —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç–∏)
    i = n - 1
    while i >= 0:
        index = arr[i] // exp1
        output[count[index % 10] - 1] = arr[i]
        count[index % 10] -= 1
        i -= 1

    # –ö–æ–ø–∏—Ä—É–µ–º output –æ–±—Ä–∞—Ç–Ω–æ –≤ arr
    for i in range(n):
        arr[i] = output[i]


# –û—Å–Ω–æ–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è Radix Sort
def radixSort(arr):

    # –ù–∞—Ö–æ–¥–∏–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç
    max1 = max(arr)

    # –°–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ –∫–∞–∂–¥–æ–º—É —Ä–∞–∑—Ä—è–¥—É
    exp = 1
    while max1 // exp >= 1:
        countingSort(arr, exp)
        exp *= 10


# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

radixSort(arr)

for i in range(len(arr)):
    print(arr[i], end=" ")
```
