Условие потерялось.

```python
N = int(input())
A = sorted(map(int, input().split()), reverse=True)

def can(k):
    if k == 0:
        return False

    counts = [1 for _ in range(k)]
    for i in range(N):
        counts[i % k] = min(A[i], counts[i % k] - 1)
        
    for i in counts:
        if i < 0:
            return False
    return True


l, r = 0, N
while l <= r:
    m = (l + r) // 2
    if can(m):
        r = m - 1
    else:
        l = m + 1

print(l)
```
